\section{Challenges}
\begin{frame}{Preprocessing \& Normalization Challenges}
    \begin{itemize}
        \item Most EEG foundation models rely on minimal or inconsistent preprocessing
        \item Impact of noise, artifacts, and outliers is poorly understood
        \item Strong variability across datasets, devices, and acquisition protocols

    \end{itemize}

        \centering

        \end{multicols}
    \end{figure}

\end{frame}

\begin{frame}{Evaluation \& Benchmarking Limitations}
    \begin{itemize}
        \item No standardized benchmarks for EEG foundation models
        \item Evaluation often limited to in-distribution settings
        \item Few studies report few-shot or zero-shot transfer
        \item Cross-dataset generalization remains underexplored
    \end{itemize}

    \begin{figure}[h]
        \centering
        \includegraphics[width=0.2\linewidth]{figures/test.png}
    \end{figure}
\end{frame}

\begin{frame}{Long-Term Context Modeling}
    \begin{itemize}
        \item EEG exhibits multiple temporal scales
        \item Millisecond-level events (e.g., ERPs) vs hour-long dynamics (e.g., sleep)
        \item Current architectures struggle to unify short- and long-range context
    \end{itemize}

    \begin{figure}[h]
        \centering
        \includegraphics[width=0.55\linewidth]{figures/eeg_bads.png}
    \end{figure}
\end{frame}

\begin{frame}{Interpretability}
    \begin{itemize}
        \item Limited focus on model interpretability
        \item Black-box behavior hinders clinical adoption
        \item Difficult to relate learned representations to neurophysiology
        \item Trust is critical for high-stakes diagnostic applications
    \end{itemize}

\end{frame}
